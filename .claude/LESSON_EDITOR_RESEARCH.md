#### **Undo/Redo**

**ะะตะฐะปะธะทะฐัะธั ัะตัะตะท BLoC:**
```dart
class EditorBloc extends Bloc<EditorEvent, EditorState> {
  final List<EditorState> _history = [];
  int _historyIndex = -1;

  void _recordState(EditorState state) {
    // Remove future states if we're in the middle
    _history.removeRange(_historyIndex + 1, _history.length);

    // Add new state
    _history.add(state);
    _historyIndex++;

    // Limit history size
    if (_history.length > 50) {
      _history.removeAt(0);
      _historyIndex--;
    }
  }

  void undo() {
    if (_historyIndex > 0) {
      _historyIndex--;
      emit(_history[_historyIndex]);
    }
  }

  void redo() {
    if (_historyIndex < _history.length - 1) {
      _historyIndex++;
      emit(_history[_historyIndex]);
    }
  }
}
```

```dart
/// ะัะธะผะตั ัะตะฐะปะธะทะฐัะธะธ ะฐะฝะธะผะฐัะธะน
class _AnimatedObjectWidgetState extends State<AnimatedObjectWidget>
    with SingleTickerProviderStateMixin {
  late AnimationController _controller;
  late Animation<double> _animation;

  @override
  void initState() {
    super.initState();
    _controller = AnimationController(
      duration: widget.enterDuration,
      vsync: this,
    );

    _animation = _buildAnimation(widget.enterEffect);
    _controller.forward();
  }

  @override
  void dispose() {
    _controller.dispose();
    super.dispose();
  }

  Animation<double> _buildAnimation(AnimationEffect effect) {
    switch (effect) {
      case AnimationEffect.appear:
        return Tween<double>(begin: 1.0, end: 1.0).animate(_controller);

      case AnimationEffect.fade:
        return Tween<double>(begin: 0.0, end: 1.0).animate(
          CurvedAnimation(parent: _controller, curve: Curves.easeIn),
        );

      case AnimationEffect.bounce:
        return Tween<double>(begin: 0.0, end: 1.0).animate(
          CurvedAnimation(parent: _controller, curve: Curves.bounceOut),
        );

      case AnimationEffect.zoom:
        return Tween<double>(begin: 0.0, end: 1.0).animate(
          CurvedAnimation(parent: _controller, curve: Curves.easeOutBack),
        );

      // ... ะดััะณะธะต ัััะตะบัั
      default:
        return Tween<double>(begin: 0.0, end: 1.0).animate(_controller);
    }
  }

  @override
  Widget build(BuildContext context) {
    return AnimatedBuilder(
      animation: _animation,
      builder: (context, child) {
        return _buildEffectWidget(widget.enterEffect, child!);
      },
      child: Text(
        widget.emoji,
        style: TextStyle(fontSize: 48),
      ),
    );
  }

  Widget _buildEffectWidget(AnimationEffect effect, Widget child) {
    switch (effect) {
      case AnimationEffect.appear:
        return child; // No animation

      case AnimationEffect.fade:
        return Opacity(opacity: _animation.value, child: child);

      case AnimationEffect.flyInBottom:
        return Transform.translate(
          offset: Offset(0, 100 * (1 - _animation.value)),
          child: Opacity(opacity: _animation.value, child: child),
        );

      case AnimationEffect.flyInLeft:
        return Transform.translate(
          offset: Offset(-100 * (1 - _animation.value), 0),
          child: Opacity(opacity: _animation.value, child: child),
        );

      case AnimationEffect.zoom:
        return Transform.scale(
          scale: _animation.value,
          child: Opacity(opacity: _animation.value, child: child),
        );

      case AnimationEffect.bounce:
        return Transform.scale(
          scale: _animation.value,
          child: child,
        );

      case AnimationEffect.swivel:
        return Transform.rotate(
          angle: (1 - _animation.value) * 2 * 3.14159,
          child: Opacity(opacity: _animation.value, child: child),
        );

      case AnimationEffect.pinwheel:
        return Transform.rotate(
          angle: (1 - _animation.value) * 4 * 3.14159,
          child: Transform.scale(
            scale: _animation.value,
            child: Opacity(opacity: _animation.value, child: child),
          ),
        );

      default:
        return child;
    }
  }
}
```
